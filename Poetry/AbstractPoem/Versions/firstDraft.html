abstract Poem() {

  if (aPoem.isShort()) {
    if (aPoem.isSweet()) {
      doesThePoemCarryAPerfectWeight();
    }
    else if (aPoem.isBitter().andDark()) {
      canItGiveLightToTheCornersOfTheMind();
    }
  }

  if (aPoem.isGrand()) {
    if (aPoem.isPrime()) {
      whoAreWeToJudgeItsMerits();
    }
    else if (aPoem.isShunned().andRejected()) {
      whoMakesTheCallAndDeniesItsExistence();
    }
  }

}
